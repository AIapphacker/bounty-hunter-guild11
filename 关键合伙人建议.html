python
import matplotlib.pyplot as plt
import numpy as np

# 画布
fig, ax = plt.subplots(figsize=(7, 7))
ax.axis('equal')

# 飞轮环节
steps = [
    "用户增长\n(留学生/移民)", 
    "任务生态\n(雇主 ↔ 猎人)", 
    "资金托管\n(USDT+智能合约)", 
    "信用体系\n(评分/信用积累)", 
    "数据&AI优化\n(推荐/审核/风控)", 
    "社区文化\n(社交/身份感)", 
    "平台收益\n(佣金/广告)"
]

# 标记护城河环节
moat_steps = {"资金托管\n(USDT+智能合约)", "信用体系\n(评分/信用积累)", "数据&AI优化\n(推荐/审核/风控)"}

# 环节位置
angles = [i * 2 * np.pi / len(steps) for i in range(len(steps))]

# 绘制节点和箭头
for i, (step, theta) in enumerate(zip(steps, angles)):
    x = 2.8 * np.cos(theta)
    y = 2.8 * np.sin(theta)
    # 区分护城河颜色
    if step in moat_steps:
        facecolor = "#FFD700"  # 金色护城河
    else:
        facecolor = "#D3E3F1"  # 淡蓝普通环节
    
    ax.text(x, y, step, ha='center', va='center', fontsize=9, weight='bold',
            bbox=dict(facecolor=facecolor, edgecolor="black", boxstyle="round,pad=0.4"))
    
    next_theta = angles[(i+1) % len(steps)]
    ax.annotate("", xy=(2.3*np.cos(next_theta), 2.3*np.sin(next_theta)),
                xytext=(2.3*np.cos(theta), 2.3*np.sin(theta)),
                arrowprops=dict(arrowstyle="->", lw=1.5))

# 中心主题
ax.text(0, 0, "Bounty Hunter Guild\n护城河+闭环飞轮",
        ha='center', va='center', fontsize=11, weight='bold',
        bbox=dict(facecolor="lightyellow", edgecolor="black", boxstyle="round,pad=0.6"))

# 添加图例
from matplotlib.patches import Patch
legend_elements = [
    Patch(facecolor="#FFD700", edgecolor="black", label="核心护城河环节"),
    Patch(facecolor="#D3E3F1", edgecolor="black", label="一般环节")
]
ax.legend(handles=legend_elements, loc='upper right', bbox_to_anchor=(1.3, 1))

ax.axis('off')

# 保存文件
output_path = "/mnt/data/flywheel_bounty_hunter_moat.png"
plt.savefig(output_path, dpi=130, bbox_inches='tight')
plt.close()
